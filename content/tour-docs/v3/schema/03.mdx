```ts title="solution"
import { Effect, Schema, Either } from "effect"

const NameSchema = Schema.String

const combineNames = (first: string, last: string) => `${first} ${last}`

const first = Schema.decodeUnknownEither(NameSchema)("Alice")
const last = Schema.decodeUnknownEither(NameSchema)("Smith")
const result = Either.zipWith(first, last, combineNames)

if (Either.isRight(result)) {
  console.log("Valid:", result.right)
} else {
  console.log("Invalid:", result.left)
}

// Or in Effect pipeline: use decode
const program = Schema.decodeUnknown(NameSchema)("Alice")
Effect.runPromise(program).then(console.log)
```

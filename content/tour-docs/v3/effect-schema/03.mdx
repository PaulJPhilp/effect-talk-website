```ts title="concept"
import { Effect, Schema } from "effect"

// API sends date as ISO string; we want a Date inside our app
// transform: decode string → Date, encode Date → string
```

```ts title="solution"
import { Effect, Schema } from "effect"

const DateFromString = Schema.transform(Schema.String, Schema.ValidDateFromSelf, {
  strict: true,
  decode: (value) => new Date(value),
  encode: (date) => date.toISOString(),
})

const EventSchema = Schema.Struct({
  name: Schema.String,
  at: DateFromString,
})

const program = Schema.decodeUnknown(EventSchema)({
  name: "Launch",
  at: "2025-06-01T12:00:00.000Z",
}).pipe(
  Effect.tap((e) => Effect.sync(() => console.log(e.at instanceof Date)))
)

Effect.runPromise(program)
```

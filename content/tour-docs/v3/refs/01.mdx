```ts title="solution"
import { Effect, Ref } from "effect"

const program = Effect.gen(function* () {
  const ref = yield* Ref.make(0)
  yield* Ref.update(ref, (n) => n + 1)
  yield* Ref.update(ref, (n) => n + 1)
  const value = yield* Ref.get(ref)
  return value
})

Effect.runPromise(program).then(console.log) // 2
```
